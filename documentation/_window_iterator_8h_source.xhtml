201  template <typename M>
202  void iterate_2D_internal(M &&on_new_row_size, int end_x, int end_y)
203  {
204  //Is there more than one row to process ?
205  if(end_y == _position.y())
206  {
207  // Single row:
208  // Both start and end belong to the same row:
209  iterate_over_dim0(end_x + _w.x().step(), on_new_row_size);
210  }
211  else
212  {
213  // Do we start from the beginning of the row ?
214  if(_w.x().start() != _position.x())
215  {
216  //Start in the middle of a row: process left-over X
217  iterate_over_dim0(_w.x().end(), on_new_row_size);
218  _position[1] += _w.y().step();
219  }
220 
221  //Middle rows
222  bool no_leftover = end_x + _w.x().step() == _w.x().end();
223  if(no_leftover)
224  {
225  //Switch to full row size:
226  on_new_row_size(_w[0].start(), _w.x().end());
227  // Shouldn't be possible to reach that point and not have at least one entire row to process
228  ARM_COMPUTE_ERROR_ON(_w.y().end() == _position.y());
229  // No leftover: all the rows lefts to process are full width:
230  iterate_over_dim1(end_y + _w.y().step());
231  }
232  else
233  {
234  // Are there full rows to process ?
235  if(_position[1] != end_y)
236  {
237  //Switch to full row size:
238  on_new_row_size(_w[0].start(), _w.x().end());
239  iterate_over_dim1(end_y);
240  }
241 
242  //Leftover end x
243  _position[0] = _w.x().start();
244  iterate_over_dim0(end_x + _w.x().step(), on_new_row_size);
245  }
246  }
247  }
