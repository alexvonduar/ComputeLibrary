322 TEST_SUITE(GEMMConvolutionLayer)
323 
324 DATA_TEST_CASE(Configuration, framework::DatasetMode::ALL, combine(combine(framework::dataset::concat(datasets::SmallConvolutionLayerDataset(), datasets::LargeConvolutionLayerDataset()),
325  CNNDataTypes),
326  framework::dataset::make("ActivationInfo",
327 { ActivationLayerInfo(ActivationLayerInfo::ActivationFunction::RELU) })),
328 input_shape, weights_shape, bias_shape, output_shape, info, dilation, data_type, act_info)
329 {
330  auto bias_data_type = is_data_type_quantized_asymmetric(data_type) ? DataType::S32 : data_type;
331 
332  // Create tensors
333  Tensor src = create_tensor<Tensor>(input_shape, data_type, 1, QuantizationInfo(2.f / 255.f, 127));
334  Tensor weights = create_tensor<Tensor>(weights_shape, data_type, 1, QuantizationInfo(2.f / 255.f, 127));
335  Tensor bias = create_tensor<Tensor>(bias_shape, bias_data_type, 1, QuantizationInfo(2.f / 255.f, 127));
336  Tensor dst = create_tensor<Tensor>(output_shape, data_type, 1, QuantizationInfo(2.f / 255.f, 127));
337 
338  ARM_COMPUTE_EXPECT(src.info()->is_resizable(), framework::LogLevel::ERRORS);
339  ARM_COMPUTE_EXPECT(weights.info()->is_resizable(), framework::LogLevel::ERRORS);
340  ARM_COMPUTE_EXPECT(bias.info()->is_resizable(), framework::LogLevel::ERRORS);
341  ARM_COMPUTE_EXPECT(dst.info()->is_resizable(), framework::LogLevel::ERRORS);
342 
343  const QuantizationInfo src_quantization_info = src.info()->quantization_info();
344  const QuantizationInfo weights_quantization_info = weights.info()->quantization_info();
345 
346  // Create and configure function
347  NEGEMMConvolutionLayer conv;
348  conv.configure(&src, &weights, &bias, &dst, info, WeightsInfo(), dilation, act_info);
349 
350  // Validate valid region
351  const ValidRegion src_valid_region = shape_to_valid_region(input_shape);
352  const ValidRegion weights_valid_region = shape_to_valid_region(weights_shape);
353  const ValidRegion bias_valid_region = shape_to_valid_region(bias_shape);
354  const ValidRegion dst_valid_region = shape_to_valid_region(output_shape);
355 
356  validate(src.info()->valid_region(), src_valid_region);
357  validate(weights.info()->valid_region(), weights_valid_region);
358  validate(bias.info()->valid_region(), bias_valid_region);
359  validate(dst.info()->valid_region(), dst_valid_region);
360 
361  // Validate QuantizationInfo
362  ARM_COMPUTE_EXPECT(src.info()->quantization_info() == src_quantization_info, framework::LogLevel::ERRORS);
363  ARM_COMPUTE_EXPECT(weights.info()->quantization_info() == weights_quantization_info, framework::LogLevel::ERRORS);
364 
365  // Validate padding
366  //TODO(COMPMID-415) Need to validate padding?
367 }
