50 void NEGEMMInterleavedWrapper::prepare()
51 {
52  if(!_is_prepared)
53  {
54  if(_pretranspose_b)
55  {
56  NEScheduler::get().schedule(_prepare_b.get(), Window::DimX);
57  _b->mark_as_unused();
58  }
59  else
60  {
61  _prepare_b->create_workloads(_b_workloads);
62  }
63  _transform_a->create_workloads(_a_workloads);
64  _matrix_multiply->create_workloads(_mm_workloads);
65 
66  //Maximum number of workloads to create:
67  const unsigned int num_threads = NEScheduler::get().num_threads();
68  const unsigned int max_iterations = num_threads == 1 ? 1 : num_threads;
69  //Maximum number of iterations the parameters allow:
70  const unsigned int num_iterations = _batch_window.num_iterations_total();
71  // Keep the smallest of the two:
72  const unsigned int num_windows = std::min(num_iterations, max_iterations);
73  const TensorShape window_shape = _batch_window.shape();
74 
75  // Create a 1D window to dynamically split the batch window:
76  Window win_1D;
77  win_1D.set(0, Window::Dimension(0, num_iterations));
78 
79  // Create one workload for each sub-window:
80  for(unsigned int w = 0; w < num_windows; w++)
81  {
82  Window win = win_1D.split_window(0, w, num_windows);
83  const Coordinates start_offset = index2coords(window_shape, win.x().start());
84  const Coordinates end_offset = index2coords(window_shape, win.x().end() - 1);
85  const unsigned int num_x_blocks = _block_walker.num_iterations(Window::DimX);
86 
87  auto workload = [start_offset, end_offset, num_x_blocks, this](const ThreadInfo & info)
88  {
89  //For each block of rows in "M"
90  auto workload_mm = this->_mm_workloads.begin();
91  for(auto workload_a = this->_a_workloads.begin(); workload_a != this->_a_workloads.end(); workload_a++)
92  {
93  // Transform one k_block from A:
94  this->_transform_a->transform(*workload_a, info, this->_batch_window, start_offset, end_offset);
95  // Then perform the matrix multiplication for each x block along N:
96  for(unsigned int i = 0; i < num_x_blocks; i++)
97  {
98  ARM_COMPUTE_ERROR_ON(workload_mm == this->_mm_workloads.end());
99  this->_matrix_multiply->transform(*workload_mm++, info, this->_batch_window, start_offset, end_offset);
100  }
101  }
102  };
103  _workloads.push_back(workload);
104  }
105 
106  _is_prepared = true;
107  }
108 }

141 void NEGEMMInterleavedWrapper::configure(const ITensor *a, const ITensor *b, ITensor *c, float alpha, float beta, bool pretranspose_b, bool use_dot)
142 {
143  _params = INEGEMMWrapperKernel::extract_parameters(a, b, c);
144  _a = a;
145  _b = b;
146  _c = c;
147  _pretranspose_b = pretranspose_b;
148 
149  DataType input_type = a->info()->data_type();
150 
151  // Forcing 128-byte alignment (required by 32-bit kernels)
152  const unsigned int alignment = 128;
153  _transformed_b.allocator()->init(TensorInfo{}, alignment);
154  _tmp_c.allocator()->init(TensorInfo{}, alignment);
155  _tag = "NEGEMMInterleaved_";
156  _tag += get_strategy_name(input_type, use_dot);
157 
158  if(!_pretranspose_b)
159  {
160  // If B is transposed at every iteration then transformed_B can be managed:
161  _memory_group.manage(&_transformed_b);
162  _block_sizes = calculate_block_sizes_from_data_type(NEScheduler::get().cpu_info(), _params.M, _params.N, _params.K, input_type, use_dot);
163  }
164  else
165  {
166  _tag += "_preB";
167  switch(input_type)
168  {
169  case DataType::F32:
170  _prepare_b = instantiate_prepareB<float>(_b, &_transformed_b, _params);
171  break;
172 #ifdef __aarch64__
173  case DataType::U8:
174  case DataType::QASYMM8:
175  if(use_dot)
176  {
177  _prepare_b = instantiate_prepareB<uint8_t, true>(_b, &_transformed_b, _params);
178  }
179  else
180  {
181  _prepare_b = instantiate_prepareB<uint8_t, false>(_b, &_transformed_b, _params);
182  }
183  break;
184  case DataType::S8:
185  if(use_dot)
186  {
187  _prepare_b = instantiate_prepareB<int8_t, true>(_b, &_transformed_b, _params);
188  }
189  else
190  {
191  _prepare_b = instantiate_prepareB<int8_t, false>(_b, &_transformed_b, _params);
192  }
193  break;
194 #endif /* __aarch64__ */
195 #ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
196  case DataType::F16:
197  _prepare_b = instantiate_prepareB<__fp16>(_b, &_transformed_b, _params);
198  break;
199 #endif /* __ARM_FEATURE_FP16_VECTOR_ARITHMETIC */
200  default:
201  ARM_COMPUTE_ERROR("DataType not supported");
202  break;
203  }
204  ARM_COMPUTE_ERROR_ON(_prepare_b == nullptr);
205 
206  _block_sizes = _prepare_b->block_sizes();
207  }
208 
209  _block_walker.set(Window::DimX, Window::Dimension(0, ceil_to_multiple(_params.N, _block_sizes.x_block), _block_sizes.x_block));
210  _block_walker.set(Window::DimY, Window::Dimension(0, ceil_to_multiple(_params.K, _block_sizes.k_block), _block_sizes.k_block));
211  _block_walker.set(Window::DimZ, Window::Dimension(0, _params.multis));
212 
213  _batch_window.set(Window::DimX, Window::Dimension(0, ceil_to_multiple(_block_sizes.m_round, _block_sizes.strategy_out_height), _block_sizes.strategy_out_height));
214  _batch_window.set(Window::DimY, Window::Dimension(0, _params.batches));
215 
216  _transformed_a.allocator()->init(TensorInfo(TensorShape{ _block_sizes.k_block, _block_sizes.m_round, _params.batches }, 1, input_type), alignment);
217  _memory_group.manage(&_transformed_a);
218  _memory_group.manage(&_tmp_c);
219 
220  switch(input_type)
221  {
222  case DataType::F32:
223  _transform_a = instantiate_transformA<float>(_a, &_transformed_a, _block_walker, _params);
224  _matrix_multiply = instantiate_matrix_multiply<float, float>(&_transformed_a, &_transformed_b, &_tmp_c, c, _block_walker, _block_sizes, _params, pretranspose_b, alpha, beta);
225  break;
226 #ifdef __aarch64__
227  case DataType::U8:
228  case DataType::QASYMM8:
229  if(use_dot)
230  {
231  _transform_a = instantiate_transformA<uint8_t, true>(_a, &_transformed_a, _block_walker, _params);
232  _matrix_multiply = instantiate_matrix_multiply<uint8_t, uint32_t, true>(&_transformed_a, &_transformed_b, &_tmp_c, c, _block_walker, _block_sizes, _params, pretranspose_b, alpha, beta);
233  }
234  else
235  {
236  _transform_a = instantiate_transformA<uint8_t, false>(_a, &_transformed_a, _block_walker, _params);
237  _matrix_multiply = instantiate_matrix_multiply<uint8_t, uint32_t, false>(&_transformed_a, &_transformed_b, &_tmp_c, c, _block_walker, _block_sizes, _params, pretranspose_b, alpha, beta);
238  }
239  break;
240  case DataType::S8:
241  if(use_dot)
242  {
243  _transform_a = instantiate_transformA<int8_t, true>(_a, &_transformed_a, _block_walker, _params);
244  _matrix_multiply = instantiate_matrix_multiply<int8_t, int32_t, true>(&_transformed_a, &_transformed_b, &_tmp_c, c, _block_walker, _block_sizes, _params, pretranspose_b, alpha, beta);
245  }
246  else
247  {
248  _transform_a = instantiate_transformA<int8_t, false>(_a, &_transformed_a, _block_walker, _params);
249  _matrix_multiply = instantiate_matrix_multiply<int8_t, int32_t, false>(&_transformed_a, &_transformed_b, &_tmp_c, c, _block_walker, _block_sizes, _params, pretranspose_b, alpha, beta);
250  }
251  break;
252 #endif /* __aarch64__ */
253 #ifdef __ARM_FEATURE_FP16_VECTOR_ARITHMETIC
254  case DataType::F16:
255  _transform_a = instantiate_transformA<__fp16>(_a, &_transformed_a, _block_walker, _params);
256  _matrix_multiply = instantiate_matrix_multiply<__fp16, __fp16>(&_transformed_a, &_transformed_b, &_tmp_c, c, _block_walker, _block_sizes, _params, pretranspose_b, alpha, beta);
257  break;
258  break;
259 #endif /* __ARM_FEATURE_FP16_VECTOR_ARITHMETIC */
260  default:
261  break;
262  }
263  ARM_COMPUTE_ERROR_ON(_transform_a == nullptr);
264  ARM_COMPUTE_ERROR_ON(_matrix_multiply == nullptr);
265  _transformed_a.allocator()->allocate();
266  _tmp_c.allocator()->allocate();
267  _transformed_b.allocator()->allocate();
268 }
