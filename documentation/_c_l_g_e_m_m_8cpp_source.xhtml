43 inline bool is_interleaved_transposed(int m, int n, int k, DataType data_type, bool reshape_b_only_on_first_run, GPUTarget gpu_target)
44 {
45  bool flag = true;
46 
47  if(gpu_target_is_in(gpu_target, GPUTarget::G52, GPUTarget::G52LIT, GPUTarget::G71, GPUTarget::G72, GPUTarget::G76))
48  {
49  // COMPMID-852
50  if(k > 256 && m > 4 && is_data_type_float(data_type) && reshape_b_only_on_first_run)
51  {
52  constexpr float alpha = 3.2f;
53  constexpr float fact0 = 1.51f;
54  constexpr float fact1 = 1.66f;
55  constexpr float ops = 12.0f;
56  const float scale = k > 1024 ? 1.07f : 1.0f;
57  flag = alpha + ((n * fact0) / ops) < ((fact1 * n * scale) / ops);
58  }
59  else
60  {
61  flag = false;
62  }
63  }
64  else
65  {
66  // We reshape the matrices only if we do not have the vector-by-matrix case and we reshape the matrix B only once
67  flag = m != 1 && reshape_b_only_on_first_run;
68  }
69 
70  return flag;
71 }

90 void CLGEMM::configure(const ICLTensor *a, const ICLTensor *b, const ICLTensor *c, ICLTensor *output, float alpha, float beta, const GEMMInfo &gemm_info)
91 {
92  ARM_COMPUTE_ERROR_ON_NULLPTR(a, b, output);
93 
94  // Perform validation step
95  ARM_COMPUTE_ERROR_THROW_ON(validate(a->info(), b->info(), c != nullptr ? c->info() : nullptr, output->info(), alpha, beta, gemm_info));
96 
97  // Check if we need to reshape the matrix B only on the first run
98  _reshape_b_only_on_first_run = gemm_info.reshape_b_only_on_first_run();
99  _is_prepared = gemm_info.retain_internal_weights();
100  _original_b = b;
101 
102  const ICLTensor *matrix_a = a;
103  const ICLTensor *matrix_b = b;
104 
105  // Get the GPU target
106  const GPUTarget gpu_target = CLScheduler::get().target();
107 
108  // Set the target for the kernels
109  _interleave_kernel.set_target(gpu_target);
110  _mm_kernel.set_target(gpu_target);
111 
112  // Arguments used by GEMMReshapeInfo
113  // If we pass the matrix A and matrix B reshaped to CLGEMMMatrixMultiplyKernel, we need to pass m, n, k, mult_transpose1xW_width and mult_interleave4x4_height to CLGEMMReshapeInfo
114  // in order to know how the matrices have been reshaped
115  bool reinterpret_input_as_3d = gemm_info.reinterpret_input_as_3d();
116  const int m = reinterpret_input_as_3d ? (a->info()->dimension(1) * a->info()->dimension(2)) : a->info()->dimension(1);
117  const int n = b->info()->dimension(0);
118  const int k = a->info()->dimension(0);
119  const int depth_output_gemm3d = gemm_info.depth_output_gemm3d();
120  int mult_transpose1xW_width = 1;
121  int mult_interleave4x4_height = 1;
122 
123  if(get_arch_from_target(gpu_target) == GPUTarget::BIFROST)
124  {
125  mult_transpose1xW_width = 4;
126  mult_interleave4x4_height = 2;
127  }
128 
129  // Check if we need to reshape the matrix A and matrix B
130  _is_interleaved_transposed = is_interleaved_transposed(m, n, k, a->info()->data_type(), _reshape_b_only_on_first_run, gpu_target);
131 
132  // if _is_interleaved_transposed is set, force reinterpret_input_as_3d to be false as the output of CLGEMMInterleaveKernel will be 2D
133  if(_is_interleaved_transposed)
134  {
135  reinterpret_input_as_3d = false;
136 
137  matrix_a = &_tmp_a;
138  matrix_b = &_tmp_b;
139 
140  // Manage intermediate buffers
141  _memory_group.manage(&_tmp_a);
142  if(!_reshape_b_only_on_first_run)
143  {
144  _memory_group.manage(&_tmp_b);
145  }
146  // _tmp_a and _tmp_b will be auto configured in _interleave_kernel and in _transpose_kernel
147 
148  // Configure interleave kernel
149  _interleave_kernel.configure(a, &_tmp_a, mult_interleave4x4_height, gemm_info.reinterpret_input_as_3d());
150 
151  // Configure transpose kernel
152  _transpose_kernel.configure(b, &_tmp_b, mult_transpose1xW_width);
153  }
154 
155  // Configure and tune matrix multiply kernel
156  _mm_kernel.configure(matrix_a, matrix_b, output, alpha, _is_interleaved_transposed, GEMMReshapeInfo(m, n, k,
157  mult_transpose1xW_width, mult_interleave4x4_height,
158  depth_output_gemm3d, reinterpret_input_as_3d),
159  gemm_info.fp_mixed_precision());
160  CLScheduler::get().tune_kernel_static(_mm_kernel);
161 
162  if(_is_interleaved_transposed)
163  {
164  // Allocate intermediate tensors
165  _tmp_a.allocator()->allocate();
166  if(!_reshape_b_only_on_first_run)
167  {
168  _tmp_b.allocator()->allocate();
169  }
170  }
171 
172  // Configure matrix addition kernel
173  if(beta != 0 && c != nullptr)
174  {
175  _ma_kernel.configure(c, output, beta);
176  _run_addition = true;
177  }
178 }

180 Status CLGEMM::validate(const ITensorInfo *a, const ITensorInfo *b, const ITensorInfo *c, const ITensorInfo *output, float alpha, float beta, const GEMMInfo &gemm_info)
181 {
182  ARM_COMPUTE_UNUSED(alpha);
183  ARM_COMPUTE_UNUSED(output);
184 
185  // Check if we need to reshape the matrix B only on the first run
186  const bool reshape_b_only_on_first_run = gemm_info.reshape_b_only_on_first_run();
187 
188  const ITensorInfo *matrix_a_info = a;
189  const ITensorInfo *matrix_b_info = b;
190 
191  TensorInfo tmp_a_info{};
192  TensorInfo tmp_b_info{};
193 
194  // Get the GPU target
195  const GPUTarget gpu_target = CLScheduler::get().target();
196 
197  // Arguments used by GEMMReshapeInfo
198  // If we pass the matrix A and matrix B reshaped to CLGEMMMatrixMultiplyKernel, we need to pass m, n, k, mult_transpose1xW_width and mult_interleave4x4_height to CLGEMMReshapeInfo
199  // in order to know how the matrices have been reshaped
200  bool reinterpret_input_as_3d = gemm_info.reinterpret_input_as_3d();
201  const int m = reinterpret_input_as_3d ? (a->dimension(1) * a->dimension(2)) : a->dimension(1);
202  const int n = b->dimension(0);
203  const int k = a->dimension(0);
204  int mult_transpose1xW_width = 1;
205  int mult_interleave4x4_height = 1;
206  const int depth_output_gemm3d = gemm_info.depth_output_gemm3d();
207 
208  if(get_arch_from_target(gpu_target) == GPUTarget::BIFROST)
209  {
210  mult_transpose1xW_width = 4;
211  mult_interleave4x4_height = 2;
212  }
213 
214  // Check if we need to reshape the matrix A and matrix B
215  const bool run_interleave_transpose = is_interleaved_transposed(m, n, k, a->data_type(), reshape_b_only_on_first_run, gpu_target);
216 
217  // if _is_interleaved_transposed is set, force reinterpret_input_as_3d to be false as the output of CLGEMMInterleaveKernel will be 2D
218  if(run_interleave_transpose)
219  {
220  reinterpret_input_as_3d = false;
221  }
222 
223  const GEMMReshapeInfo reshape_info = GEMMReshapeInfo(m, n, k, mult_transpose1xW_width, mult_interleave4x4_height, depth_output_gemm3d, reinterpret_input_as_3d);
224 
225  if(run_interleave_transpose)
226  {
227  matrix_a_info = &tmp_a_info;
228  matrix_b_info = &tmp_b_info;
229 
230  // Validate interleave kernel
231  auto_init_if_empty(tmp_a_info, a->clone()->set_tensor_shape(compute_interleaved_shape(*a, mult_interleave4x4_height, gemm_info.reinterpret_input_as_3d())));
232  ARM_COMPUTE_RETURN_ON_ERROR(CLGEMMInterleave4x4Kernel::validate(a, &tmp_a_info, mult_interleave4x4_height, gemm_info.reinterpret_input_as_3d()));
233 
234  // Validate transpose kernel
235  auto_init_if_empty(tmp_b_info, b->clone()->set_tensor_shape(compute_transpose1xW_with_element_size_shape(*b, mult_transpose1xW_width)));
236  ARM_COMPUTE_RETURN_ON_ERROR(CLGEMMTranspose1xWKernel::validate(b, &tmp_b_info, mult_transpose1xW_width));
237  }
238 
239  // Validate matrix multiply
240  ARM_COMPUTE_RETURN_ON_ERROR(CLGEMMMatrixMultiplyKernel::validate(matrix_a_info, matrix_b_info, output, alpha, run_interleave_transpose, reshape_info, gpu_target, gemm_info.fp_mixed_precision()));
241 
242  if(beta != 0 && c != nullptr)
243  {
244  // Validate matrix addition kernel
245  ARM_COMPUTE_RETURN_ON_ERROR(CLGEMMMatrixAdditionKernel::validate(c, output, beta));
246  }
247 
248  return Status{};
249 }

251 void CLGEMM::run()
252 {
253  prepare();
254 
255  _memory_group.acquire();
256 
257  if(_is_interleaved_transposed)
258  {
259  // Run interleave kernel
260  CLScheduler::get().enqueue(_interleave_kernel, false);
261 
262  if(!_reshape_b_only_on_first_run)
263  {
264  // Run transpose kernel
265  CLScheduler::get().enqueue(_transpose_kernel, false);
266  }
267  }
268 
269  // Run matrix multiply kernel
270  CLScheduler::get().enqueue(_mm_kernel, !_run_addition);
271 
272  // Run matrix addition kernel
273  if(_run_addition)
274  {
275  CLScheduler::get().enqueue(_ma_kernel);
276  }
277 
278  _memory_group.release();
279 }

281 void CLGEMM::prepare()
282 {
283  if(!_is_prepared)
284  {
285  if(_is_interleaved_transposed && _reshape_b_only_on_first_run)
286  {
287  // Run transpose kernel and mark original weights tensor as unused
288  _tmp_b.allocator()->allocate();
289  CLScheduler::get().enqueue(_transpose_kernel, false);
290  _original_b->mark_as_unused();
291  }
292  CLScheduler::get().queue().finish();
293  _is_prepared = true;
294  }
